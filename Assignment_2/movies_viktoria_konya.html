GET APIs of 100 movies

# Get APIs
get_api <- function(url) {
  
  t <- GET(url)
  t <- fromJSON(content(t, "text"))
  
  # ID
  id <- t$id
  
  # Production companies
  production_companies <- paste0(t$production_companies$name, collapse = ", ")
  
  # Budget
  runtime <- t$runtime
  
  # Genres
  genres <- paste0(t$genres$name, collapse = ", ")

  # Title
  title <- t$original_title
  
  # Overview
  overview <- t$overview
  
  # Budget
  budget <- t$budget
  
  # Revenue
  revenue <- t$revenue
  
  # Popularity
  popularity <- t$popularity
  
  # Vote Average
  vote_average <- t$vote_average
  
  # Vote Count
  vote_count <- t$vote_count
  
  # Df
  df <- data.frame(id, 
                   title, 
                   production_companies, 
                   runtime, budget, 
                   revenue, 
                   popularity, 
                   vote_average, 
                   vote_count, 
                   overview)
  
  colnames(df) <- c("ID", 
                    "Title", 
                    "Production companies",  
                    "Runtime", 
                    "Budget", 
                    "Revenue",
                    "Popularity", 
                    "Vote Average", 
                    "Vote Count", 
                    "Overview")
  
  return(df)
  
}
